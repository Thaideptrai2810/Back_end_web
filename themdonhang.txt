// Lấy danh sách đơn hàng từ localStorage, nếu không có sẽ trả về mảng rỗng
let orders = JSON.parse(localStorage.getItem('orders')) || [];

// Tạo đơn hàng mới
let newOrder = {
    id: '#DH' + (orders.length + 1), // Tăng ID đơn hàng
    date: "23/11/2024", // Ngày đơn hàng
    name: "Nguyễn Văn B", // Tên khách hàng
    address: "Quận 3", // Địa chỉ giao hàng
    total: "150,000 ₫", // Tổng tiền
    status: "processing", // Trạng thái đơn hàng
    statusText: "Chưa giao hàng", // Trạng thái hiển thị
    actionText: "Xử lý" // Hành động xác nhận
};

// Thêm đơn hàng vào danh sách
orders.push(newOrder);

// Lưu lại danh sách đơn hàng vào localStorage
localStorage.setItem('orders', JSON.stringify(orders));

// Hiển thị thông báo thành công
console.log("Đơn hàng mới đã được thêm vào Local Storage.");
